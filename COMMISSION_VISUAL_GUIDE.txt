┌─────────────────────────────────────────────────────────────────────────┐
│  TABLE: commission_records                                              │
├─────────────────────────────────────────────────────────────────────────┤
│  id                    INT (Primary Key)                                │
│  transaction_id        INT (Foreign Key → payment_transactions)         │
│  booking_id            INT (Foreign Key → vendor_bookings)              │
│  vendor_id             INT (Foreign Key → vendors)                      │
│  transaction_amount    DECIMAL(12,2)                                    │
│  commission_rate       DECIMAL(5,2) [Default: 2.50]                     │
│  commission_amount     DECIMAL(12,2) [Calculated]                       │
│  status                ENUM('Pending','Collected','Waived')             │
│  collected_date        TIMESTAMP                                        │
│  collected_by          INT (Foreign Key → users)                        │
│  payment_method        ENUM('Cash','Bank Transfer','UPI','Other')      │
│  reference_number      VARCHAR(100)                                     │
│  notes                 TEXT                                             │
│  created_at            TIMESTAMP                                        │
│  updated_at            TIMESTAMP                                        │
└─────────────────────────────────────────────────────────────────────────┘

┌─────────────────────────────────────────────────────────────────────────┐
│  TABLE: commission_summary                                              │
├─────────────────────────────────────────────────────────────────────────┤
│  id                    INT (Primary Key)                                │
│  vendor_id             INT (Foreign Key → vendors)                      │
│  month                 INT (1-12)                                       │
│  year                  INT                                              │
│  total_transactions    DECIMAL(12,2)                                    │
│  total_commission      DECIMAL(12,2)                                    │
│  collected_commission  DECIMAL(12,2)                                    │
│  pending_commission    DECIMAL(12,2)                                    │
│  transaction_count     INT                                              │
│  last_updated          TIMESTAMP                                        │
└─────────────────────────────────────────────────────────────────────────┘
